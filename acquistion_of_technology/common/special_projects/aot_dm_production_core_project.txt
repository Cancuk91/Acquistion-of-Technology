@research_cost = 30000
special_project = {
	key = "RESEARCH_DM_PRODUCTION_CORE_100_SUCCESS"
	cost = @research_cost
	tech_department = physics_technology
	picture = GFX_evt_physics_research
	event_chain = aot_dm_production_core_chain
	event_scope = planet_event
	location = no
	abort_trigger = {
	}
	on_success = {
		owner = {
			random_list = {
				100 = {
					add_event_chain_counter = {
						event_chain = aot_dm_production_core_chain
						counter = dm_production_core_energy_required
						amount = 50
					}
					change_variable = {
						which = current_dm_production_core_energy_required
						value = -50
					}
				}
				0 = {
					add_event_chain_counter = {
						event_chain = aot_dm_production_core_chain
						counter = dm_production_core_energy_required
						amount = -50
					}
					change_variable = {
						which = current_dm_production_core_energy_required
						value = 50
					}
				}
			}
			### SUCCESS !!?!?!?!
			IF = {
				limit = {
					check_variable = {
						which = current_dm_production_core_energy_required
						value <= 0
					}
				}
				country_event = {
					id = aot_dm_production_core_events.4
				}
			}
			### INSUFFICIENT POWER, BASE DEFENSES OFFLINE!
			ELSE_IF = {
				limit = {
					check_variable = {
						which = current_dm_production_core_energy_required
						value > 2500
					}
				}
				country_event = {
					id = aot_dm_production_core_events.5
				}
			}
			### INCOMPLETED, WE NEED MORE PYLONS!
			ELSE_IF = {
				limit = {
					check_variable = {
						which = current_dm_production_core_energy_required
						value <= 2500 
					}
					check_variable = {
						which = current_dm_production_core_energy_required
						value > 0
					}
				}
				country_event = {
					id = aot_dm_production_core_events.3
				}
			}
		}
	}
	on_fail = {
	}
}
