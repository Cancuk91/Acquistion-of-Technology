has_enough_dark_matter_for_project = {
	hidden_trigger = {
		NOT = {
			has_country_flag = has_ongoing_production_core_study
		}
	}
	if = {
		limit = {
			check_variable = {
				which = aot_production_core_project_cost
				value = 2500
			}
		}
		resource_stockpile_compare = {
			resource = sr_dark_matter
			value >= 2500
		}
	}
	if = {
		limit = {
			check_variable = {
				which = aot_production_core_project_cost
				value = 5000
			}
		}
		resource_stockpile_compare = {
			resource = sr_dark_matter
			value >= 5000
		}
	}
	if = {
		limit = {
			check_variable = {
				which = aot_production_core_project_cost
				value = 7500
			}
		}
		resource_stockpile_compare = {
			resource = sr_dark_matter
			value >= 7500
		}
	}
	if = {
		limit = {
			check_variable = {
				which = aot_production_core_project_cost
				value = 10000
			}
		}
		resource_stockpile_compare = {
			resource = sr_dark_matter
			value >= 10000
		}
	}
	if = {
		limit = {
			check_variable = {
				which = aot_production_core_project_cost
				value = 12500
			}
		}
		resource_stockpile_compare = {
			resource = sr_dark_matter
			value >= 12500
		}
	}
}
